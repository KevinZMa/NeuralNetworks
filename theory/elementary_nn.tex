\section{Logical AND}
\[
  \{0,1\}^2 \to \{0,1\}
\]

\begin{align*}
  U_0 &= \id\left(
    \begin{bmatrix}
      x^{(0)}_{1} \\
      x^{(0)}_{2}
  \end{bmatrix}\right) \\
  U_1 &=\relu\left(
    \begin{bmatrix}
      \id\left( x^{(0)}_{1} \right) + \id\left( x^{(0)}_{2} \right) - 1
    \end{bmatrix}
  \right)
\end{align*}

\section{Logical OR}

\[
  \{0,1\}^2 \to \{0,1\}
\]

Let $S$ be the sum of the inputs, which is $x^{(0)}_1 + x^{(0)}_2$.
We output whether $S>0$ by the taking the ReLU
of $S$ and $S-1$. If $S\le 0$, then $0-0=0$. If $S>0$, then $S-(S-1)=1$.

\begin{align*}
  U_0 &= \id\left(
    \begin{bmatrix}
      x^{(0)}_{1} \\
      x^{(0)}_{2}
  \end{bmatrix}\right) \\
  U_1 &= \relu\left(
    \begin{bmatrix}
      \id\left( x^{(0)}_{1} + x^{(0)}_{2} \right) \\
      \id\left( x^{(0)}_{1} + x^{(0)}_{2} \right) - 1
  \end{bmatrix}\right) \\
  U_2 &= \id\left(
    \begin{bmatrix}
      \relu\left( \id\left( x^{(0)}_{1} + x^{(0)}_{2} \right) \right) - \relu\left( \id\left( x^{(0)}_{1} + x^{(0)}_{2} \right) - 1 \right)
    \end{bmatrix}
  \right)
\end{align*}

    \end{bmatrix}$}
    node[below=3pt] at (\p1) {$\textbf{b}^{2}=
      \begin{bmatrix}0
    \end{bmatrix}$};
  \end{tikzpicture}
\end{center}
